```js
const num = 3.141592653589793238;

console.log("Floor ", Math.floor(num));
//소수점 첫째 자리에서 내림(버림). 3.1은 3, -1.1은 -2가 됩니다.
console.log("Ceil ", Math.ceil(num));
//소수점 첫째 자리에서 올림. 3.1은 4, -1.1은 -1이 됩니다.
console.log("Round ", Math.round(num));
//소수점 첫째 자리에서 반올림. 3.1은 3, 3.6은 4, -1.1은 -1이 됩니다.
console.log("Trunc ", Math.trunc(num)) //Internet Explorer에서는 지원하지 않음
//소수부를 무시. 3.1은 3이 되고 -1.1은 -1이 됩니다.

console.log(Math.random())
//0과 1 사이의 난수를 반환합니다(1은 제외).

Math.max(10, 13, 25) / Math.min(25, 0.5, -45)
//인수 중 최대/최솟값을 반환합니다.

console.log("Max ", Math.max(3, 5, -10, 0, 1)); // 5
console.log("Min ", Math.min(1, 2)); // 1

console.log("Power ", Math.pow(2, 8))
//n을 power번 거듭제곱한 값을 반환합니다.

console.log(Math.pow(2, 10)); // 2의 10제곱 = 1024
```
